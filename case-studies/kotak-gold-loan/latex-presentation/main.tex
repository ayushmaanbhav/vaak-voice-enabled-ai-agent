% Stat card
\newcommand{\statcard}[3]{%
    \begin{tikzpicture}
        \node[minimum width=2.8cm, minimum height=1.8cm, fill=kotaklightgray, draw=kotakgold, line width=0.5pt, rounded corners=5pt] (card) {};
        \node[font=\fontsize{20}{24}\selectfont\bfseries, text=kotakred] at ([yshift=0.25cm]card.center) {#1};
        \node[font=\footnotesize, text=kotakblue] at ([yshift=-0.25cm]card.center) {#2};
        \node[font=\tiny, text=kotakgray] at ([yshift=-0.55cm]card.center) {#3};
    \end{tikzpicture}
}

% Progress bar
\newcommand{\progressbar}[3]{%
    % #1 = label, #2 = percentage, #3 = color
    \begin{tikzpicture}
        \node[anchor=east, font=\small, text width=2.5cm, align=right] at (0,0) {#1};
        \fill[kotaklightgray] (0.2,-0.15) rectangle (5.2,0.15);
        \fill[#3] (0.2,-0.15) rectangle ({0.2 + #2/100*5},0.15);
        \node[anchor=west, font=\small\bfseries, text=#3] at (5.4,0) {#2\%};
    \end{tikzpicture}
}
